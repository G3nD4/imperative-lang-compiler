routine comprehensiveRoutine(a: integer, b: integer, c: integer): integer is
    var sum: integer is a + b + c
    var product: integer is a * b * c
    var average: integer
    if sum > 100 then
        average := sum / 3
    else
        average := (a + b) / 2
    end
    var finalResult: integer is 0
    for i in 1..average loop
        if i % 2 = 0 then
            finalResult := finalResult + i
        else
            finalResult := finalResult - i
        end
    end
    while finalResult < 100 loop
        finalResult := finalResult + product
        if finalResult > 200 then
            break
        end
    end
    return finalResult
end
